def getLargestNumber(num):
    num_list=list(map(int,num))
    n=len(num_list)
    changed=True
    
    
    while changed:
        changed=False
        for i in range(n-1):
            if num_list[i] % 2 == num_list[i+1] % 2:
                if num_list[i]<num_list[i+1]:
                    num_list[i],num_list[i+1]=num_list[i+1],num_list[i]
                    changed=True
    return ''.join(map(str,num_list))
